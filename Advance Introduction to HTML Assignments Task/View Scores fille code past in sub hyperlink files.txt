        // This is the file whose source code will be loaded and displayed.
        const sourceFilePath = './Complete missing part.html'; 

        function displaySourceCode() {
            const outputElement = document.getElementById('sourceCodeOutput');
            outputElement.innerText = "Loading source code...";

            fetch(sourceFilePath)
                .then(response => {
                    if (!response.ok) {
                        throw new Error(`HTTP error! Status: ${response.status}`);
                    }
                    return response.text(); // Get the response as raw text
                })
                .then(htmlContent => {
                    // Display the raw HTML text (not rendered)
                    outputElement.innerText = htmlContent; 
                })
                .catch(error => {
                    outputElement.innerText = `Error loading file: ${error.message}`;
                    console.error('Fetch error:', error);
                });
        }
    </script>`
